f_daily_data(symbol) =>
    [current_price, pct_change, prior_close, prior_high, prior_low] = 
         request.security(symbol, 'D', 
         [close, 
         math.round(( (close - close[1]) / close[1] ) * 100, 2),
         close[1],
         high[1],
         low[1]
         ])
